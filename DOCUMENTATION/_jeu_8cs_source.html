114 protected virtual void validBtn_Click(object sender, EventArgs e)
115 {
116 Player currentPlayer = IsFirstPlayer ? J1 : J2;
117
118 // Si exo 3, on prend la valeur des deux txtBox
119 if (IndexExercice == 3)
120 {
121 repTxtBox.Text = GetRepTxtBox(1);
122 }
123 // Si dernier exercice, on prend la valeur du radio btn comme réponse.
124 else if (IndexExercice == 5)
125 {
126 repTxtBox.Text = GetRepTxtBox(2);
127 }
128 // Sinon
129 else repTxtBox.Text = GetRepTxtBox();
130
131 // On process la réponse
132 ShowAnswerResult(currentPlayer, Exercices[IndexExercice], repTxtBox.Text);
133
134 // Pour passer du joueur 1 au joueur 2, on change de suite sur le même type d'exercice
135 if (IsFirstPlayer) Exercices[IndexExercice].GetNewSuite(TypeSuite);
136 // Sinon on change de type d'exercice
137 else IndexExercice++;
138 // On change le tour
139 IsFirstPlayer = !IsFirstPlayer;
140
141 // Si on est pas sur le dernier exercice et que le tour est au joueur 2
142 if (!IsFirstPlayer && IndexExercice <= 5)
143 {
144 ResetAffichage(J2, Exercices[IndexExercice]);
145 }
146 // Si on a terminé
147 else if (IndexExercice == 6)
148 {
149 EndGame();
150 }
151 // Sinon on reset l'affichage et montre les radio buttons en cas d'exercice sur la monotonie
152 // ou les deux txt box en cas d'exercice raison et premier terme
153 else
154 {
155 ResetAffichage(J1, Exercices[IndexExercice]);
156 if (IndexExercice == 3) SwitchInputType(1);
157 else if (IndexExercice == 5) SwitchInputType(2);
158 else SwitchInputType();
159 }

222 protected void SwitchInputType(byte type = 0)
223 {
224 bool isOneChecked = false;
225 // Si RadioButtons
226 if (type == 2)
227 {
228 // On cache et desactive la TextBox
229 repTxtBox.Visible = false;
230 repTxtBox.Enabled = false;
231
232 foreach (Control btn in repBox.Controls)
233 {
234 // Si le control est un RadioButton
235 if (btn is System.Windows.Forms.RadioButton)
236 {
237 // Activé et visible
238 btn.Visible = true;
239 btn.Enabled = true;
240 // On en check un pour enlever la possibilité de valider sans réponse
241 if (!isOneChecked)
242 {
243 ((System.Windows.Forms.RadioButton)btn).Checked = true;
244 isOneChecked = !isOneChecked;
245 }
246 }
247 }
248 }
249 // Si TextBox -> 2x TextBox
250 else if (type == 1)
251 {
252 // On cache le champ de réponse de base
253 repTxtBox.Visible = false;
254 repTxtBox.Enabled = false;
255 reponseLbl.Visible = false;
256
257 // On montre le champ raison et premier terme
258 raisonTxtBox.Visible = true;
259 raisonTxtBox.Enabled = true;
260 raisonLbl.Visible = true;
261 premierTermeTxtBox.Visible = true;
262 premierTermeTxtBox.Enabled = true;

265 // Si TextBox de base
266 else
267 {
268 repTxtBox.Visible = true;
269 repTxtBox.Enabled = true;
270 reponseLbl.Visible = true;
271
272 // On cache les labels et txtBox de raison et premier terme (qu'importe leur état)

274 raisonTxtBox.Visible = false;
275 raisonTxtBox.Enabled = false;
276 premierTermeLbl.Visible = false;
